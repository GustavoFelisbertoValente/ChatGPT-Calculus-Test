\documentclass{article}
\usepackage{amsmath}
\usepackage{amsfonts}
\usepackage{amssymb}
\usepackage{graphicx}

\begin{document}

\title{Calculus Exam}
\author{Your Name}
\date{Date}

\maketitle

\section{Instructions}

\begin{itemize}
\item This exam contains \textbf{five} questions, each worth \textbf{20} points.
\item Write your solutions clearly and legibly.
\item Show all your work and justify your answers.
\item You may use a calculator and the provided formula sheet.
\item The exam is \textbf{closed book} except for the provided formula sheet.
\item Good luck!
\end{itemize}

\section{Questions}

\subsection*{Question 1}
Find the derivative of the function $f(x) = 3x^4 - 4x^3 + 2x - 5$.

\subsection*{Question 2}
Find the limit $\lim_{x \to 0} \frac{\sin(x)}{x}$.

\subsection*{Question 3}
Find the area between the curves $y = x^2$ and $y = 2x - x^2$.

\subsection*{Question 4}
Evaluate the integral $\int_1^3 \frac{3x-1}{x^2+2x+1} dx$.

\subsection*{Question 5}
Find the equation of the tangent line to the curve $y = e^{2x}\cos(x)$ at the point where $x = \frac{\pi}{4}$.

\section{Formula Sheet}

\begin{itemize}
\item Power rule: $\frac{d}{dx}(x^n) = nx^{n-1}$.
\item Product rule: $\frac{d}{dx}(f(x)g(x)) = f'(x)g(x) + f(x)g'(x)$.
\item Quotient rule: $\frac{d}{dx}\left(\frac{f(x)}{g(x)}\right) = \frac{f'(x)g(x) - f(x)g'(x)}{(g(x))^2}$.
\item Chain rule: $\frac{d}{dx}(f(g(x))) = f'(g(x))g'(x)$.
\item Integration by substitution: $\int f(g(x))g'(x)dx = \int f(u)du$, where $u = g(x)$.
\item Integration by parts: $\int u dv = uv - \int v du$.
\item Fundamental theorem of calculus: $\int_a^b f(x)dx = F(b) - F(a)$, where $F(x)$ is an antiderivative of $f(x)$.
\item Trigonometric identities: $\sin^2(x) + \cos^2(x) = 1$, $\sin(2x) = 2\sin(x)\cos(x)$, $\cos(2x) = \cos^2(x) - \sin^2(x)$.
\end{itemize}

\end{document}
